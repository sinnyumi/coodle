        <button onclick="startGame()" style="padding: 10px 20px; font-size: 1rem; background-color: #2563eb; color: white; border: none; border-radius: 5px; cursor: pointer;">대결 입장</button>

    <!-- 3. 교사용 실시간 자동 순위 전광판 -->
    <div id="admin-container" class="hidden">
        <h1>🏆 실시간 Kordle 배틀 순위판 (자동 연동) 🏆</h1>
        <p style="color: #6366f1; font-weight: bold;">새로고침 안 해도 학생들이 맞히면 실시간으로 등수가 바뀝니다!</p>
        <button onclick="clearDatabase()" style="background-color: #ef4444; color: white; border: none; padding: 5px 10px; border-radius: 4px; cursor: pointer;">기록 초기화(새 게임)</button>
        <table>
            <thead>
                <tr><th>순위</th><th>이름</th><th>시도 횟수</th><th>소요 시간</th></tr>
            </thead>
            <tbody id="leaderboard-rows"></tbody>
        </table>

    <script type="module">
        // 초간단 실시간 무료 DB 연동 (선생님이 세팅할 필요 없음!)
        import { Datashare } from 'https://cdn.jsdelivr.net/npm/datashare-client@1.0.1/dist/datashare-client.js';
        
        // 고유한 교실 방 ID 자동 생성 (선생님 깃허브 주소 기반)
        const roomID = btoa(window.location.hostname + window.location.pathname).replace(/=/g, '');
        const share = new Datashare(roomID);

        // 💡 [교사 정답 설정] 정확히 6칸 자모음 분리 (예: '공책' -> ㄱㅗㅇㅊㅐㄱ)
        const ANSWER = ['ㄱ', 'ㅗ', 'ㅇ', 'ㅊ', 'ㅐ', 'ㄱ']; 
        
        let nickname = "";
        let startTime, timerInterval;
        let currentRow = 0;
        let currentTile = 0;
        let guesses = Array(6).fill().map(() => Array(6).fill(""));

        // 전광판 화면 전환 규칙 설정
        const urlParams = new URLSearchParams(window.location.search);
        if (urlParams.get('admin') === 'true') {
            document.getElementById('login-container').className = 'hidden';
            document.getElementById('admin-container').className = 'hidden';
            document.getElementById('admin-container').classList.remove('hidden');
            
            // 데이터 실시간 감시 시작
            share.on('change', (data) => {
                updateLeaderboardView(data);
            });
            // 최초 데이터 수신
            updateLeaderboardView(share.get());
        }

        // 보드판 및 키보드 초기화
        const board = document.getElementById('board');
        for (let i = 0; i < 6; i++) {
            const row = document.createElement('div'); row.className = 'row';
            for (let j = 0; j < 6; j++) {
                const tile = document.createElement('div'); tile.className = 'tile'; tile.id = `tile-${i}-${j}`; row.appendChild(tile);
            }
            board.appendChild(row);
        }
        const keys = [['ㄱ','ㄴ','ㄷ','ㄹ','ㅁ','ㅂ','ㅅ','ㅇ','ㅈ','ㅊ','ㅋ','ㅌ','ㅍ','ㅎ'],['ㅏ','ㅑ','ㅓ','ㅕ','ㅗ','ㅛ','ㅜ','ㅠ','ㅡ','ㅣ','ㅐ','ㅔ'],['ENTER','BACK']];
        const keyboard = document.getElementById('keyboard');
        keys.forEach(row => {
            const keyRow = document.createElement('div'); keyRow.className = 'key-row';
            row.forEach(key => {
                const button = document.createElement('button'); button.className = 'key'; button.textContent = key;
                button.onclick = () => window.handleKeyPress(key); keyRow.appendChild(button);
            });
            keyboard.appendChild(keyRow);
        });

        window.startGame = function() {
            nickname = document.getElementById('nickname-input').value.trim();
            if (!nickname) return alert("이름을 입력해 주세요!");
            document.getElementById('login-container').classList.add('hidden');
            document.getElementById('game-container').classList.remove('hidden');
            document.getElementById('player-info').textContent = `${nickname} 님 플레이 중...`;
            startTime = new Date();
            timerInterval = setInterval(() => {
                document.getElementById('timer').textContent = `시간: ${Math.floor((new Date() - startTime) / 1000)}초`;
            }, 1000);
        }

        window.handleKeyPress = function(key) {
            if (key === 'BACK' && currentTile > 0) {
                currentTile--; guesses[currentRow][currentTile] = ""; document.getElementById(`tile-${currentRow}-${currentTile}`).textContent = "";
            } else if (key === 'ENTER') {
                if (currentTile === 6) checkRow(); else alert("6글자를 다 채우세요!");
            } else if (currentTile < 6 && key !== 'BACK' && key !== 'ENTER') {
                guesses[currentRow][currentTile] = key; document.getElementById(`tile-${currentRow}-${currentTile}`).textContent = key; currentTile++;
            }
        }

        function checkRow() {
            const currentGuess = guesses[currentRow];
            let correctCount = 0;
            for (let i = 0; i < 6; i++) {
                const tile = document.getElementById(`tile-${currentRow}-${i}`);
                if (ANSWER[i] === currentGuess[i]) { tile.classList.add('correct'); correctCount++; }
                else if (ANSWER.includes(currentGuess[i])) tile.classList.add('present');
                else tile.classList.add('absent');
            }
            if (correctCount === 6) {
                clearInterval(timerInterval);
                const finalTime = Math.floor((new Date() - startTime) / 1000);
                
                // 🚀 자동으로 실시간 데이터 전송!
                let currentResults = share.get() || {};
                currentResults[nickname] = { name: nickname, attempts: currentRow + 1, time: finalTime };
                share.set(currentResults);
                
                alert(`축하합니다! 정답입니다!\n기록이 교실 전광판에 즉시 자동 등록되었습니다.`);
            } else {
                currentRow++; currentTile = 0;
                if (currentRow === 6) { clearInterval(timerInterval); alert("실패했습니다! 다음 기회에.."); }
            }
        }

        function updateLeaderboardView(data) {
            const tbody = document.getElementById('leaderboard-rows');
            tbody.innerHTML = "";
            if (!data) return;
            
            let players = Object.values(data);
            players.sort((a, b) => a.attempts !== b.attempts ? a.attempts - b.attempts : a.time - b.time);
            
            players.forEach((p, idx) => {
                tbody.innerHTML += `<tr>
                    <td>${idx + 1}등 🏆</td>
                    <td>${p.name}</td>
                    <td>${p.attempts}회</td>
                    <td>${p.time}초</td>
                </tr>`;
            });
        }

        window.clearDatabase = function() {
            if(confirm("정말 모든 순위 기록을 초기화하시겠습니까?")) {
                share.set({});
            }
        }
    </script>
