        <button onclick="startGame()" style="padding: 10px 20px; font-size: 1rem; background-color: #2563eb; color: white; border: none; border-radius: 5px; cursor: pointer;">게임 시작!</button>

        <h3 id="player-info"></h3>

        <div class="grid" id="board"></div>
        <div id="keyboard"></div>
    </div>

    <!-- 3. 교사용 실시간 순위 전광판 -->
    <div id="admin-container" class="hidden">
        <h1>🏆 실시간 한글 워들(꼬들) 순위판 🏆</h1>
        <p style="color: #aaa;">학생이 성공 창을 교사에게 보여주면, 교사가 아래에 이름을 입력해 확정합니다.</p>
        <input type="text" id="success-player" placeholder="성공한 학생 이름 입력" style="padding: 10px; font-size: 1.1rem;">
        <input type="number" id="success-attempts" placeholder="시도 횟수" style="padding: 10px; font-size: 1.1rem; width: 80px;">
        <button onclick="addLeaderboard()" style="padding: 10px 20px; font-size: 1.1rem; background-color: #538d4e; color: white; border: none; cursor: pointer;">순위 등록</button>
        
        <table>
            <thead>
                <tr><th>순위</th><th>이름</th><th>시도 횟수</th><th>소요 시간</th></tr>
            </thead>
            <tbody id="leaderboard-rows"></tbody>
        </table>

    <script>
        // 💡 [교사 설정 구역] 오늘의 정답 단어를 자모음으로 분리해서 적어주세요! (정확히 6칸이어야 합니다)
        // 예: '기말고사' -> ㄱㅣㅁㅏㄹㄱㅗㅅㅏ (X, 8칸이라 안됨)
        // 추천: 3글자 받침 없는 단어 (나무위키 -> ㄴㅏㅁㅜㅇㅜㅣ 6칸) 또는 2글자 받침 있는 단어 (수업 -> ㅅㅜㅇㅓㅂㅂ 6칸 채우기 등)
        // 여기서는 예시로 '공부하자' -> ㄱㅗㅇㅂㅜㅎㅏㅈㅏ 총 8칸이므로, 딱 6칸짜리 단어인 "과과학ㄱ" (ㄱㅘㄱㅘㅎㅏㄱ) 등으로 세팅합니다.
        const ANSWER = ['ㄱ', 'ㅗ', 'ㅇ', 'ㅊ  ', 'ㅐ', 'ㄱ']; // 예시 정답: 공책 (ㄱㅗㅇㅊㅐㄱ - 6글자)
        
        let nickname = "";
        let startTime, timerInterval;
        let currentRow = 0;
        let currentTile = 0;
        let guesses = Array(6).fill().map(() => Array(6).fill(""));

        // 주소창에 ?admin=true 가 붙어있으면 교사용 전광판으로 변신!
        const urlParams = new URLSearchParams(window.location.search);
        if (urlParams.get('admin') === 'true') {
            document.getElementById('login-container').classList.add('hidden');
            document.getElementById('admin-container').classList.remove('hidden');
            updateLeaderboardView();
        }

        // 보드판 생성
        const board = document.getElementById('board');
        for (let i = 0; i < 6; i++) {
            const row = document.createElement('div');
            row.className = 'row';
            for (let j = 0; j < 6; j++) {
                const tile = document.createElement('div');
                tile.className = 'tile';
                tile.id = `tile-${i}-${j}`;
                row.appendChild(tile);
            }
            board.appendChild(row);
        }

        // 키보드 생성 (한글 자모음)
        const keys = [
            ['ㄱ', 'ㄴ', 'ㄷ', 'ㄹ', 'ㅁ', 'ㅂ', 'ㅅ', 'ㅇ', 'ㅈ', 'ㅊ', 'ㅋ', 'ㅌ', 'ㅍ', 'ㅎ'],
            ['ㅏ', 'ㅑ', 'ㅓ', 'ㅕ', 'ㅗ', 'ㅛ', 'ㅜ', 'ㅠ', 'ㅡ', 'ㅣ', 'ㅐ', 'ㅔ'],
            ['ENTER', 'BACK']
        ];
        const keyboard = document.getElementById('keyboard');
        keys.forEach(row => {
            const keyRow = document.createElement('div');
            keyRow.className = 'key-row';
            row.forEach(key => {
                const button = document.createElement('button');
                button.className = 'key';
                button.textContent = key;
                button.onclick = () => handleKeyPress(key);
                keyRow.appendChild(button);
            });
            keyboard.appendChild(keyRow);
        });

        function startGame() {
            nickname = document.getElementById('nickname-input').value.trim();
            if (!nickname) return alert("이름을 입력해 주세요!");
            document.getElementById('login-container').classList.add('hidden');
            document.getElementById('game-container').classList.remove('hidden');
            document.getElementById('player-info').textContent = `${nickname} 님의 플레이 화면`;
            
            startTime = new Date();
            timerInterval = setInterval(() => {
                const now = new Date();
                const diff = Math.floor((now - startTime) / 1000);
                document.getElementById('timer').textContent = `시간: ${diff}초`;
            }, 1000);
        }

        function handleKeyPress(key) {
            if (key === 'BACK') {
                if (currentTile > 0) {
                    currentTile--;
                    guesses[currentRow][currentTile] = "";
                    document.getElementById(`tile-${currentRow}-${currentTile}`).textContent = "";
                }
            } else if (key === 'ENTER') {
                if (currentTile === 6) {
                    checkRow();
                } else {
                    alert("6글자를 모두 채워야 합니다!");
                }
            } else {
                if (currentTile < 6) {
                    guesses[currentRow][currentTile] = key;
                    document.getElementById(`tile-${currentRow}-${currentTile}`).textContent = key;
                    currentTile++;
                }
            }
        }

        function checkRow() {
            const currentGuess = guesses[currentRow];
            let correctCount = 0;

            for (let i = 0; i < 6; i++) {
                const tile = document.getElementById(`tile-${currentRow}-${i}`);
                const letter = currentGuess[i];

                if (ANSWER[i] === letter) {
                    tile.classList.add('correct');
                    correctCount++;
                } else if (ANSWER.includes(letter)) {
                    tile.classList.add('present');
                } else {
                    tile.classList.add('absent');
                }
            }

            if (correctCount === 6) {
                clearInterval(timerInterval);
                const finalTime = Math.floor((new Date() - startTime) / 1000);
                alert(`🎉 정답입니다! 🎉\n소요시간: ${finalTime}초\n시도횟수: ${currentRow + 1}회\n\n선생님 화면에 등록을 요청하세요!`);
            } else {
                currentRow++;
                currentTile = 0;
                if (currentRow === 6) {
                    clearInterval(timerInterval);
                    alert("아쉽게도 실패했습니다! 정답을 다시 추리해 보세요.");
                }
            }
        }

        // [교사용 시스템] 로컬스토리지 활용 실시간 스코어보드
        function addLeaderboard() {
            const name = document.getElementById('success-player').value.trim();
            const attempts = document.getElementById('success-attempts').value;
            if (!name || !attempts) return alert("이름과 시도 횟수를 적어주세요!");

            // 학생 화면에 떠 있는 시간 측정값을 확인하는 대안 (현장 구두 확인용 혹은 로컬 수동 입력)
            const timeSeconds = prompt(`${name} 학생의 화면에 표시된 '소요 시간(초)'을 입력하세요:`);
            
            let ranks = JSON.parse(localStorage.getItem('kordle_ranks') || '[]');
            ranks.push({ name, attempts: parseInt(attempts), time: parseInt(timeSeconds) });
            
            // 순위 정렬 규칙: 시도 횟수 적은 순 -> 시간 짧은 순
            ranks.sort((a, b) => a.attempts !== b.attempts ? a.attempts - b.attempts : a.time - a.time);
            
            localStorage.setItem('kordle_ranks', JSON.stringify(ranks));
            updateLeaderboardView();
            
            document.getElementById('success-player').value = "";
            document.getElementById('success-attempts').value = "";
        }

        function updateLeaderboardView() {
            const ranks = JSON.parse(localStorage.getItem('kordle_ranks') || '[]');
            const tbody = document.getElementById('leaderboard-rows');
            tbody.innerHTML = "";
            ranks.forEach((p, idx) => {
                tbody.innerHTML += `<tr>
                    <td>${idx + 1}등</td>
                    <td>${p.name}</td>
                    <td>${p.attempts}회</td>
                    <td>${p.time}초</td>
                </tr>`;
            });
        }
    </script>
